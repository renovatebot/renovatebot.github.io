    workarounds:bitnamiDockerImageVersioning¶

    Use custom regex versioning for bitnami images

    {
  "packageRules": [
    {
      "matchCurrentValue": "/^(?<major>\\d+)(?:\\.(?<minor>\\d+)(?:\\.(?<patch>\\d+))?)?-(?<compatibility>.+)-(?<build>\\d+)(?:-r(?<revision>\\d+))?$/",
      "matchDatasources": [
        "docker"
      ],
      "matchPackageNames": [
        "bitnami/**",
        "docker.io/bitnami/**",
        "gcr.io/bitnami-containers/**",
        "*-docker.pkg.dev/vmw-app-catalog/**"
      ],
      "versioning": "regex:^(?<major>\\d+)(?:\\.(?<minor>\\d+)(?:\\.(?<patch>\\d+))?)?(:?-(?<compatibility>.+)-(?<build>\\d+)(?:-r(?<revision>\\d+))?)?$"
    }
  ]
}

    workarounds:containerbase¶

    Add some containerbase overrides.

    {
  "packageRules": [
    {
      "description": "Use node versioning for `(containerbase|renovate)/node` images",
      "matchDatasources": [
        "docker"
      ],
      "matchPackageNames": [
        "/^(?:(?:docker|ghcr)\\.io/)?(?:containerbase|renovate)/node$/"
      ],
      "versioning": "node"
    }
  ]
}

    workarounds:disableEclipseLifecycleMapping¶

    Disable Eclipse m2e lifecycle-mapping placeholder package.

    {
  "packageRules": [
    {
      "enabled": false,
      "matchDatasources": [
        "maven"
      ],
      "matchPackageNames": [
        "org.eclipse.m2e:lifecycle-mapping"
      ]
    }
  ]
}

    workarounds:disableGradleReplacements¶

    Disable package replacement updates for gradle dependencies.

    {
  "packageRules": [
    {
      "enabled": false,
      "matchManagers": [
        "gradle"
      ],
      "matchUpdateTypes": [
        "replacement"
      ]
    }
  ]
}

    workarounds:disableMavenParentRoot¶

    Avoid version fetching for Maven packages detected as project root.

    {
  "packageRules": [
    {
      "enabled": false,
      "matchDepTypes": [
        "parent-root"
      ],
      "matchManagers": [
        "maven"
      ]
    }
  ]
}

    workarounds:doNotUpgradeFromAlpineStableToEdge¶

    Do not upgrade from Alpine stable to edge.

    {
  "packageRules": [
    {
      "allowedVersions": "<20000000",
      "matchCurrentVersion": "!/^\\d{8}$/",
      "matchDatasources": [
        "docker"
      ],
      "matchDepNames": [
        "alpine"
      ]
    },
    {
      "allowedVersions": "<20000000",
      "matchCurrentVersion": "!/^\\d{8}$/",
      "matchDatasources": [
        "docker"
      ],
      "matchPackageNames": [
        "alpine"
      ]
    }
  ]
}

    workarounds:ignoreHttp4sDigestMilestones¶

        "org.http4s:**"
      ]
    }
  ]
}

    workarounds:javaLTSVersions¶

    {
  "packageRules": [
    {
      "allowedVersions": "/^(?:8|11|17|21|25)(?:\\.|-|$)/",
      "description": "Limit Java runtime versions to LTS releases. To receive all major releases add `workarounds:javaLTSVersions` to the `ignorePresets` array.",
      "matchDatasources": [
        "docker",
        "java-version"
      ],
      "matchPackageNames": [
        "eclipse-temurin",
        "amazoncorretto",
        "adoptopenjdk",
        "openjdk",
        "java",
        "java-jre",
        "sapmachine",
        "/^azul/zulu-openjdk/",
        "/^bellsoft/liberica-openj(dk|re)-/",
        "/^cimg/openjdk/"
      ],
      "versioning": "regex:^(?<major>\\d+)?(\\.(?<minor>\\d+))?(\\.(?<patch>\\d+))?([\\._+](?<build>(\\d\\.?)+)(LTS)?)?(-(?<compatibility>.*))?$"
    },
    {
      "allowedVersions": "/^(?:8|11|17|21|25)(?:\\.|-|$)/",
      "description": "Limit Java runtime versions to LTS releases. To receive all major releases add `workarounds:javaLTSVersions` to the `ignorePresets` array.",
      "matchDatasources": [
        "docker",
        "java-version"
      ],
      "matchDepNames": [
        "eclipse-temurin",
        "amazoncorretto",
        "adoptopenjdk",
        "openjdk",
        "java",
        "java-jre",
        "sapmachine"

    {

        "docker"

        "bellsoft/liberica-runtime-container"
      ]
    }
  ]

    },
    {
      "description": "Liberica JDK that can be used to create a custom runtime with a help of jlink tool",

  ]
